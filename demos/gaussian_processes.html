Brownian motion

First lets start by simulating Brownian motion, as the scaling limit of simple random walks.

First lets simulate a simple random walk.

n <- 1002
coins <- sample(c(-1,1), n, replace=TRUE)
plot(cumsum(coins), type='l')

Lets look at how this rescales to look like Brownian motion.

srw_bm <- function (tmax, N) {
    coins <- sample(c(-1,1), floor(tmax * N), replace=TRUE)
    return(cumsum(coins) / sqrt(N))
}
tmax <- 3
Nvals <- c(10, 40, 160, 640)
for (N in Nvals) {
    srw <- srw_bm(tmax, N=N)
    tvals <- seq(0, tmax, length.out=length(srw))
    if (N == Nvals[1]) {
        plot(tvals, srw, type='l', xlab='time', ylab='space', ylim=c(-2*sqrt(3),2*sqrt(3)))
    } else {
        lines(tvals, srw, col=match(N, Nvals))
    }
}

Lets test this is working right. If this is a good approximation to Brownian motion, the distribution at time 3 should be Normal(0,\(\sigma^2=3\)).

srw_reps <- replicate(200, srw_bm(3, 10000))
qqnorm(srw_reps[nrow(srw_reps),])
qqline(srw_reps[nrow(srw_reps),])

That looks about right. And the sample variance is 2.5270133, which is close to 3.

Function space

We will now simulate Brownian motion on \([0,1]\) using the Levy construction, using the integrals of the Haar basis and the isometry of \(L^2([0,1])\).

haar_integral <- function (x, n, k) {
    a <- 2 * k / 2^n
    b <- (2 * k  + 1)/ 2^n
    c <- (2 * k  + 2)/ 2^n
    2^((n-1)/2) * ifelse(x < a, 0,
                         ifelse(x < b, x - a,
                                ifelse(x < c, c - x, 0)))
}

curve(haar_integral(x, n=3, k=1), from=0, to=1, ylab='Schauder function value')
curve(haar_integral(x, n=5, k=6), from=0, to=1, add=TRUE, col='red')
curve(haar_integral(x, n=5, k=12), from=0, to=1, add=TRUE, col='blue')

Ok, thats our (integrated) basis functions. Now we need to add them up with independent Gaussian coefficients.

haar_bm <- function (nmax, Z, B0 = 0, bridge=FALSE) {
  ## this function returns the nmax partial sum
  ## of B_t, as a function of t in [0,1]
    if (bridge) {
        nk <- data.frame(n=1, k=0) # if bridge, omit first term
    } else {
        nk <- data.frame(n=c(0,1), k=c(0,0))
    }
    # indices to sum over
    for (nn in seq(2, nmax)) {
        nk <- rbind(nk, data.frame(n=nn, k=0:(2^(nn-1)-1)))
    }
    # get list of integrated basis functions (as functions of upper limit of integration)
    fns <- lapply(1:nrow(nk), function (j)
                  function (x) haar_integral(x, n=nk$n[j], k=nk$k[j]))
    # the random coefficients
    if (missing(Z)) {
        Z <- rnorm(nrow(nk), 0, sd=1)
    }
    return(function (x) {
        B0 + Z %*% do.call(rbind, lapply(fns, function (f) f(x)))
    })
}

B <- haar_bm(10)
tt <- seq(0,1,length.out=1e3)
plot(tt, B(tt), col='red', type = 'l')

Lets see how we constructed \(B\) from successively rougher functions.

# get collection of independent iid N(0,1)
Z <- with(environment(B), Z)

Gaussian process regression

Now lets suppose we have noisy measurements of \(B_0 + B_t\), where \(B_0 \sim \operatorname{Normal}(0, \eta^2)\), at some time points: for \(t_1 < \cdots < t_n\) let \(X_i = B_0 + B_{t_i} + W_i\), for iid \(\operatorname{Normal}(0, \sigma^2)\) random variables \(W_i\). Now we want to sample from the distribution on \(B_0 + B_t\) given \((X_1, \ldots, X_n)\). This is equivalent to sampling from the distribution on \((B_0,Z)\) given \((X_1, \ldots, X_n)\), then using the random coefficients to obtain \(B_t\).

n <- 30 # number of data points
sigma <- 0.1 # residual sd
eta <- 0.5 # starting point sd
ti <- sort(runif(n))
B0 <- rnorm(1, sd=eta)
X <- B0 + sin(ti*3*pi) + rnorm(n, sd=sigma)
plot(ti, X)

We will want to

  1. evaluate the conditional mean at a bunch of time points, to get a smooth curve going near our data, and

  2. sample from the conditional distribution to get an idea of uncertainty.

To do this, we need the (a) covariance matrix of \(X\), and (b) the covariance matrix of \((B_0, Z)\), and (c) the covariance matrix of \(X\) with \((B_0, Z)\). Math on the board gives us that \(\operatorname{cov}[X_i, B_0] = \eta^2\), that \(\operatorname{cov}[X_i, X_j] = \eta^2 + \min(t_i, t_j) + \delta_{ij} \sigma^2\), and that \(\operatorname{cov}[X_i, Z_{m\ell}] = H_{m,\ell}(t_i)\), where \(H_{m,\ell}(t) = \int_0^t h_{m,\ell}(s)\,ds\) is the corresponding integrated basis function.

# construct covX
covX <- diag(n) * sigma^2 
for (i in 1:n) {
    for (j in 1:n) {
        covX[i,j] <- covX[i,j] + min(ti[i], ti[j]) + eta^2
    }
}
nmax <- with(environment(B), nmax)
# get basis functions from B to build covXZ
basis_fns <- with(environment(B), fns)
covXZ <- matrix(NA, nrow=n, ncol=length(basis_fns)+1) 
for (i in 1:n) {
    for (j in 1:length(basis_fns)) {
        covXZ[i,j+1] <- basis_fns[[j]](ti[i])
    }
}
covXZ[,1] <- eta^2
# construct covZZ
covZZ <- diag(length(basis_fns)+1)
covZZ[1,1] <- eta^2

The mean will be \(\Sigma_{ZX} \Sigma_{XX}^{-1} X\) and the covariance matrix will be \(\Sigma_{ZZ} - \Sigma_{ZX} \Sigma_{XX}^{-1} \Sigma_{XZ}\).

mean_mat <- t(covXZ) %*% solve(covX)
cov_mat <- covZZ - t(covXZ) %*% solve(covX, covXZ)
chol_cov_mat <- chol(cov_mat) # for sampling

mean_B <- function (x) {
  ## this function returns a list with two elements,
  ## the first being the conditional mean path,
  ## the second being the conditional mean B0
    mean_Z <- as.vector(mean_mat %*% x)
    return( haar_bm(nmax, mean_Z[-1], B0=mean_Z[1]) )
}

# these will be *deviations* from the mean
conditional_B <- function (x) {
    Z <- as.vector(chol_cov_mat %*% rnorm(length(basis_fns)+1))
    dB <- haar_bm(nmax, Z[-1], B0=Z[1])
    return( dB )
}

mB <- mean_B(X)
cB <- replicate(100, conditional_B(X))

Here we plot the conditional mean (in red) and a bunch of samples (in black).

# conditional mean points
mBvals <- as.vector(mB(tt))
